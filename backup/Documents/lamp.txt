LAMP
----

mkdir ~/www

echo "<?php phpinfo(); ?>" > ~/www/index.php

sudo -i

apt-get -y install mysql-server

    Wachtwoord 2 keer opgeven

service mysql start

apt-get -y install apache2 libapache2-mod-php7.0 php7.0-dev phpmyadmin

  (apache kiezen en 2 keer wachtwoord voor phpmyadmin opgeven)

sed -i 's/var\/www/home\/hgj/g'              /etc/apache2/apache2.conf
sed -i 's/var\/www\/html/home\/hgj\/www/g'   /etc/apache2/sites-enabled/000-default.conf
sed -i 's/RUN_USER=www-data/RUN_USER=hgj/g'  /etc/apache2/envvars

sudo pecl install ibm_db2

    /opt/db2 opgeven

echo "extension=ibm_db2.so" > /etc/php/7.0/apache2/conf.d/db2.ini
echo "ibm_db2.instance_name=db2" >> /etc/php/7.0/apache2/conf.d/db2.ini

xed /etc/init.d/apache2

onderstaande regel toevoegen vlak voor "$APACHE2CTL start"

. /var/db2/sqllib/db2profile

service apache2 stop
service apache2 start
systemctl daemon-reload

En dan moeten onderstaande 3 URL's werken.

http://localhost
http://localhost/server-status
http://localhost/phpmyadmin
